private final ReentrantLock lock = new ReentrantLock();
private final Condition notFull = lock.newCondition();
public void produce(int item) throws InterruptedException {
    lock.lock();
    try {
        while (full) notFull.await(); // ...
    } finally { lock.unlock(); }
}
